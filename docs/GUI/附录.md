```python
import pathlib, os

    return {**os.environ, "PYTHONPATH": str(PROJECT_ROOT)}
```

## 4. 临时目录管理

使用约定式临时目录 `{原文件所在目录}/.wb_optimize/`，优化完成后不清理（对比模式需引用产物）。应用关闭时由 MainWindow 清理。

```python
import os, shutil

class TempDirManager:
    """管理 .wb_optimize/ 临时目录"""

    def __init__(self, source_path: str):
        self._tmpdir = os.path.join(os.path.dirname(source_path), ".wb_optimize")

    @property
    def path(self) -> str:
        os.makedirs(self._tmpdir, exist_ok=True)
        return self._tmpdir

    def cleanup(self):
        """应用关闭时清理"""
        if self._tmpdir and os.path.exists(self._tmpdir):
            shutil.rmtree(self._tmpdir, ignore_errors=True)

    def find_merged(self, source_path: str) -> str | None:
        """约定式查找 _merged.json，供 ComparePanel 自动检测使用"""
        base = os.path.splitext(os.path.basename(source_path))[0]
        merged = os.path.join(self._tmpdir, f"{base}_merged.json")
        return merged if os.path.exists(merged) else None
```
