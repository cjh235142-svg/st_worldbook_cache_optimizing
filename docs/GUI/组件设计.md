### 2.2 层级 3 条目的双栏映射

| 状态 | 左栏（原始） | 右栏（映射） |
|------|-------------|-------------|
| 未优化 | `条目名 uid=N order=M` | `—`（空） |
| 分析后（可拆分） | ☐ `条目名 uid=N order=M` | `建议拆分` |
| 拆分后 | ☑ `条目名 uid=N order=M` | `→ [静]子条目 [动]子条目` |
| 优化后—保持 | `条目名 uid=N order=M` | `→ order:N→N', pos:N→N'` |
| 优化后—合并 | `条目名 uid=N order=M` | `→ 合并入 组X (合并:min-max)` |
| 优化后—拆分 | `条目名 uid=N order=M` | `→ 静态→组X、动态→uid=N'` |
| 新建（副本/包裹） | `—` | `→ 管线自动生成` |

### 2.3 层级 4 条目详情

条目节点展开后，显示以下子节点（纯文本，每个子节点一行）：

```
├─ ▸ 条目A [uid=1]
│    ├─ content: <御坂美琴>\n## 人物介绍\n学园都市...
│    ├─ key: ["御坂美琴"]
│    ├─ keysecondary: []
│    ├─ order: 100
│    ├─ position: 0 (before)
│    ├─ depth: 4

│    ├─ disable: false
```

层级 4 节点不需要双栏渲染，使用默认 `QStyledItemDelegate` 单栏显示。

```python
ROLE_RIGHT_TEXT = Qt.UserRole + 6  # 右栏文本 (str)
ROLE_CHECKABLE = Qt.UserRole + 7   # 是否可拆分勾选 (bool)
ROLE_CHECKED = Qt.UserRole + 8     # 勾选状态 (Qt.Checked / Qt.Unchecked)
ROLE_ICON = Qt.UserRole + 9        # 条目类型图标 (str: "🔵"|"🟢"|"📄")
```

### 2.5 折叠控制

```python
class WorldBookTree(QTreeWidget):
    def collapse_to_level(self, level: int):
        """展开到指定层级，折叠更低层级。"""
        it = QTreeWidgetItemIterator(self)
        while it.value():
            item = it.value()
            item_level = item.data(0, ROLE_LEVEL) or 0
            item.setExpanded(item_level < level)
            ++it
```

## 3. DualColumnDelegate

```python
class DualColumnDelegate(QStyledItemDelegate):
    """
    左右 50% 分栏绘制器。
    - level < 3: 不做分栏，调用 super().paint()
    - level == 3: 左右分栏 + CheckBox + 图标 + 右栏着色
    - level == 4: 不做分栏，调用 super().paint()

    图标约定（条目类型）：
    - 🔵 (蓝色圆圈) constant=true
    - 🟢 (绿灯) 动态条目
    - 📄 (文档) 混合/静态且非蓝灯

    颜色约定:
    - 合并: 右栏蓝底 (#E3F2FD)
    - 拆分: 右栏黄底 (#FFF9C4)
    - 自动生成: 右栏绿底 (#E8F5E9)
    - 默认: 右栏浅灰 (#F5F5F5)
    """

    checkStateChanged = pyqtSignal(int, bool)  # uid, checked

    def paint(self, painter, option, index):
        level = index.data(ROLE_LEVEL)
        if level is None or level != 3:
            super().paint(painter, option, index)
            return

        painter.save()
        rect = option.rect
        w = rect.width()
        h = rect.height()
        left_w = w // 2
        right_w = w - left_w

        # 选中态背景
        if option.state & QStyle.State_Selected:
            painter.fillRect(rect, option.palette.highlight())

        # 分割线
        painter.setPen(QPen(QColor("#ccc"), 1))
        painter.drawLine(rect.x() + left_w, rect.y(),
                         rect.x() + left_w, rect.y() + h)

        # 左栏: CheckBox + 图标 + 文本
        checkbox_w = 20
        icon_w = 20
        text_x = rect.x() + checkbox_w + icon_w + 4
        text_w = left_w - checkbox_w - icon_w - 8
        if index.data(ROLE_CHECKABLE):
            cb_opt = QStyleOptionButton()
            cb_opt.rect = QRect(rect.x() + 2, rect.y(), checkbox_w, h)
            is_checked = (index.data(ROLE_CHECKED) == Qt.Checked)
            cb_opt.state = (QStyle.State_On if is_checked
                            else QStyle.State_Off) | QStyle.State_Enabled
            self.parent().style().drawControl(QStyle.CE_CheckBox, cb_opt, painter)
        # 图标
        icon_char = index.data(ROLE_ICON) or ""
        if icon_char:
            painter.setPen(QColor("#555"))
            painter.drawText(QRect(rect.x() + checkbox_w + 2, rect.y(), icon_w, h),
                             Qt.AlignCenter, icon_char)
        left_text = index.data(ROLE_LEFT_TEXT) or ""
        painter.setPen(option.palette.windowText().color())
        painter.drawText(QRect(text_x, rect.y(), text_w, h),
                         Qt.AlignLeft | Qt.AlignVCenter, left_text)

        # 右栏: 着色 + 文本
        right_text = index.data(ROLE_RIGHT_TEXT) or ""
        right_rect = QRect(rect.x() + left_w + 4, rect.y(), right_w - 4, h)
        bg = self._color_for_right_text(right_text)
        painter.fillRect(right_rect, bg)
        painter.setPen(QColor("#333"))
        painter.drawText(right_rect, Qt.AlignLeft | Qt.AlignVCenter, right_text)
        painter.restore()

    def _color_for_right_text(self, text: str) -> QColor:
        if "合并" in text:     return QColor("#E3F2FD")
        if "拆分" in text:     return QColor("#FFF9C4")
        if "自动生成" in text: return QColor("#E8F5E9")
        return QColor("#F5F5F5")

    def editorEvent(self, event, model, option, index):
        """处理 CheckBox 点击（仅勾选框区域），发射 checkStateChanged 信号。"""
        if not index.data(ROLE_CHECKABLE):
            return False
        if event.type() == QEvent.MouseButtonRelease:
            cb = QRect(option.rect.x() + 2, option.rect.y(), 20, option.rect.height())
            if cb.contains(event.pos()):
                checked = (index.data(ROLE_CHECKED) == Qt.Checked)
                new_state = Qt.Unchecked if checked else Qt.Checked
                model.setData(index, new_state, ROLE_CHECKED)
                uid = index.data(ROLE_UID)
                if uid is not None:
                    self.checkStateChanged.emit(uid, new_state == Qt.Checked)
                return True
        return False
```

```python
class ComparePanel(QWidget):
    """
    对比面板，提供两种子模式。
    条目级 diff：复用树的右栏，旁加说明文字。
    全文级 diff：独立 FulltextDiffView。

    自动检测优化后文件：切换到对比模式时扫描 {文件目录}/.wb_optimize/{文件名}_merged.json。
    """

    def __init__(self):
        self.mode_group = QButtonGroup(self)
        self.entry_btn = QRadioButton("条目级 diff")
        self.fulltext_btn = QRadioButton("全文级 diff")
        self.stacked = QStackedWidget()

        # 状态0: 无优化后文件提示
        self.no_file_widget = QWidget()
        no_file_layout = QVBoxLayout()
        no_file_layout.addWidget(QLabel("未检测到优化结果，请先执行优化"))
        browse_btn = QPushButton("浏览选择优化后文件...")
        browse_btn.clicked.connect(self._browse_optimized)
        no_file_layout.addWidget(browse_btn)
        no_file_layout.addStretch()
        self.no_file_widget.setLayout(no_file_layout)

        # 状态1: 条目级 diff 说明
        self.entry_label = QLabel("条目级 diff 显示在左侧树的右栏中\n展开节点即可查看映射关系，右栏着色区分类型")
        # 状态2: 全文级 diff
        self.fulltext_view = FulltextDiffView()
        self.inner_stacked = QStackedWidget()
        self.inner_stacked.addWidget(self.entry_label)       # index 0
        self.inner_stacked.addWidget(self.fulltext_view)     # index 1

        self.stacked.addWidget(self.no_file_widget)          # page 0
        self.stacked.addWidget(self.inner_stacked)           # page 1

        self.mode_group.buttonClicked.connect(self._on_mode_switch)
        layout = QVBoxLayout()
        layout.addLayout(self._build_mode_bar())
        layout.addWidget(self.stacked)
        self.setLayout(layout)

    def _on_mode_switch(self, btn):
        if self.stacked.currentIndex() == 0:
            return  # 无优化文件时不允许切换
        self.inner_stacked.setCurrentIndex(0 if btn == self.entry_btn else 1)

    def check_for_optimized(self, source_path: str):
        """约定式检测：在 .wb_optimize/ 目录下查找 _merged.json"""
        import os
        base = os.path.splitext(os.path.basename(source_path))[0]
        tmp_dir = os.path.join(os.path.dirname(source_path), ".wb_optimize")
        merged = os.path.join(tmp_dir, f"{base}_merged.json")
        if os.path.exists(merged):
            self._optimized_path = merged
            self.set_optimized_ready(True)
        else:
            self._optimized_path = None
            self.set_optimized_ready(False)

    def set_optimized_ready(self, ready: bool):
        self.stacked.setCurrentIndex(1 if ready else 0)

    def _browse_optimized(self):
        path, _ = QFileDialog.getOpenFileName(self, "选择优化后世界书", "", "JSON (*.json)")
        if path:
            self._optimized_path = path
            self.set_optimized_ready(True)
```

## 7. FulltextDiffView（全文 diff）

使用单个 `QTextBrowser` 渲染 `difflib.HtmlDiff.make_file()` 生成的完整双栏 HTML。

```python
class FulltextDiffView(QWidget):
    def __init__(self):
        self.browser = QTextBrowser()
        self.browser.setReadOnly(True)
        layout = QVBoxLayout()
        layout.addWidget(self.browser)
        self.setLayout(layout)

    def load_diff(self, left_text: str, right_text: str):
        differ = difflib.HtmlDiff(tabsize=2)
        html = differ.make_file(
            left_text.splitlines(keepends=True),
            right_text.splitlines(keepends=True),
            fromdesc="原始", todesc="优化后",
            context=True, numlines=3,
        )
        self.browser.setHtml(html)
```

### 分步优化流程

使用 `PipelineWorker` 串行执行，连接信号进行进度追踪：

```
OptimizePanel 信号连接:
  worker = PipelineWorker(input_path, tmp_dir, wrapper_name)
  worker.step_done.connect(self._on_step_done)          # 更新进度条 + 步骤图标
  worker.step_error.connect(self._on_step_error)         # 弹错误对话框
  worker.pipeline_finished.connect(self._on_finished)    # 加载摘要 + 启用覆盖按钮
  worker.pipeline_cancelled.connect(self._on_cancelled)  # 恢复按钮状态
```
