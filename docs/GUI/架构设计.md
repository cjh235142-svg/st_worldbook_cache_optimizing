## 2. 模块职责

### 核心组件

| 模块 | 职责 |
|------|------|
| `main.py` | 创建 QApplication，加载样式表，实例化 MainWindow |
| `main_window.py` | 主窗口：水平 QSplitter（左树 + 右面板），菜单栏/工具栏/状态栏。连接 `WorldBookTree._on_file_dropped` 信号实现拖拽打开 `.json` |

### 数据模型

| 模块 | 职责 |
|------|------|
| `models/world_book_model.py` | 加载 JSON 到内存、按 6 键排序、分类（静态/动态/混合）、提供条目查询和树节点构建所需数据。**仅存数据，不涉及 UI**。树中所有 position 组统一使用 static/dynamic/mixed 分类，pos=4 不做 depth 子分组 |

`WorldBookModel` 完整 API：

```python
class WorldBookModel:
    def load(self, path: str) -> None
        """加载 JSON 文件到内存"""

    def get_entry(self, uid: int) -> dict | None
        """按 UID 精确查找条目"""

    def get_entries(self, pos: int, category: str) -> list[dict]
        """按 position 和类别（static/dynamic/mixed）获取条目列表"""

    def get_entries_by_filter(self, filter: str) -> list[dict]
        """按过滤条件获取：'static' | 'dynamic' | 'mixed' | 'all'"""

    def get_position_groups(self) -> list[int]
        """返回有数据的 position 值列表"""

    def search(self, text: str) -> list[dict]
        """在 comment 或 content 中模糊搜索"""

    def update_entries(self, entries: list[dict]) -> None
        """批量更新条目（拆分/优化后调用），触发内部排序"""

    def get_stats(self) -> dict
        """返回 {total, static, dynamic, mixed}"""

    def classify_entry(self, content: str) -> str
        """返回 'static' | 'dynamic' | 'mixed'"""

    def sort_key(self, entry: dict) -> tuple
        """返回 6 键排序元组 (category, sub_order, stability, position, depth_key, _original_order)"""
```

## 3. 树组件 (`WorldBookTree`) — 4 级节点

### 节点层级

| 层级 | 内容 | 创建时机 |
|------|------|---------|
| 0 | 根节点（文件名） | 初始化时 |
| 1 | position 分组（pos=0, pos=1, ...），仅创建有数据的组 | 初始化时 |
| 2 | 类别分组（静态/动态/混合），所有 position 统一 | 初始化时 |
| 3 | 具体条目（comment + uid） | 展开层级 2 时懒加载 |
| 4 | 条目字段详情（content 全文、key、order 等） | 展开层级 3 时懒加载 |

> 所有 position 组（含 pos=4）统一使用 static/dynamic/mixed 作为层级 2 类别，不做 depth≥10/depth<10 子分组。

### 懒加载策略

```python
class WorldBookTree(QTreeWidget):
    """
    懒加载树：初始只构建根节点和层级 0-2。
    用户展开某节点时按需创建子节点。
    """

    def __init__(self, model: WorldBookModel):
        self._model = model
        super().__init__()
        self.setAcceptDrops(True)  # 支持拖拽打开
        self.itemExpanded.connect(self._on_expand)
        self.itemCollapsed.connect(self._on_collapse)  # 折叠时销毁 widget
        self._build_root_levels()

    # ── 拖拽打开 ──
    def dragEnterEvent(self, event):
        if event.mimeData().hasUrls() and any(u.toLocalFile().endswith('.json') for u in event.mimeData().urls()):
            event.acceptProposedAction()

    def dropEvent(self, event):
        for url in event.mimeData().urls():
            path = url.toLocalFile()
            if path.endswith('.json'):
                self._on_file_dropped.emit(path)  # 自定义信号，MainWindow 连接后加载
                break

    def _build_root_levels(self):
        """构建根节点 → position 分组 (层级0-1) → 类别 (层级2)。只构建有数据的 position 组。"""
        for pos in self._model.get_position_groups():
            pos_item = QTreeWidgetItem([f"position={pos}"])
            pos_item.setData(0, ROLE_LEVEL, 1)
            pos_item.setChildIndicatorPolicy(QTreeWidgetItem.ShowIndicator)
            for category in ('static', 'dynamic', 'mixed'):
                cat_labels = {'static': '静态', 'dynamic': '动态', 'mixed': '混合'}
                cat_item = QTreeWidgetItem([cat_labels[category]])
                cat_item.setData(0, ROLE_LEVEL, 2)
                cat_item.setData(0, ROLE_POS, pos)
                cat_item.setData(0, ROLE_CATEGORY, category)
                cat_item.setChildIndicatorPolicy(QTreeWidgetItem.ShowIndicator)
                pos_item.addChild(cat_item)
            self.addTopLevelItem(pos_item)

    def _on_expand(self, item):
        level = item.data(0, ROLE_LEVEL)
        if level == 2:
            self._populate_entries(item)
        elif level == 3:
            self._populate_entry_detail(item)

    def _on_collapse(self, item):
        """折叠条目节点时销毁 QTextBrowser 以释放内存"""
        if item.data(0, ROLE_LEVEL) == 3:
            for i in range(item.childCount()):
                child = item.child(i)
                if child.data(0, ROLE_LEVEL) == 4:
                    self.removeItemWidget(child, 0)

    def _populate_entries(self, cat_item):
        """类别节点展开 → 创建条目节点 (层级3)"""
        if cat_item.childCount() > 0:
            return  # 防止重复创建
        pos = cat_item.data(0, ROLE_POS)
        category = cat_item.data(0, ROLE_CATEGORY)
        for entry in self._model.get_entries(pos, category):
            entry_item = QTreeWidgetItem()
            entry_item.setText(0, entry['comment'])
            entry_item.setData(0, ROLE_LEVEL, 3)
            entry_item.setData(0, ROLE_UID, entry['uid'])
            entry_item.setData(0, ROLE_LEFT_TEXT, self._left_text(entry))
            entry_item.setData(0, ROLE_RIGHT_TEXT, "—")
            entry_item.setChildIndicatorPolicy(QTreeWidgetItem.ShowIndicator)
            # 可拆分条目显示 CheckBox（使用自定义 ROLE_CHECKED，不设 ItemIsUserCheckable）
            if entry.get('suggested_split'):
                entry_item.setData(0, ROLE_CHECKABLE, True)
                entry_item.setData(0, ROLE_CHECKED, Qt.Unchecked)
            cat_item.addChild(entry_item)

    def _populate_entry_detail(self, entry_item):
        """条目节点展开 → 创建字段详情节点 (层级4)
        - content 子节点: 嵌入 QTextBrowser 显示全文（只读）
        - key/order/position 等字段: 单行 QTreeWidgetItem 显示
        """
        if entry_item.childCount() > 0:
            return
        uid = entry_item.data(0, ROLE_UID)
        entry = self._model.get_entry(uid)
        content = entry.get('content', '')
        # content 子节点: 空占位节点 + 内嵌 QTextBrowser
        content_placeholder = QTreeWidgetItem(["content"])
        content_placeholder.setData(0, ROLE_LEVEL, 4)
        # 按 80 字换行估算高度（比按行数更准确）
        estimated_lines = max(1, len(content) // 80) + content.count('\n')
        content_placeholder.setSizeHint(0, QSize(0, min(300, estimated_lines * 16)))
        entry_item.addChild(content_placeholder)
        # 展开占位节点后才嵌入 QTextBrowser（避免一次性创建过多 widget）
        tree = entry_item.treeWidget()
        if tree:
            browser = QTextBrowser()
            browser.setReadOnly(True)
            browser.setPlainText(content)
            browser.setFrameShape(QTextBrowser.NoFrame)
            tree.setItemWidget(content_placeholder, 0, browser)
        # 其他字段: 单行显示
        for field_name in ('key', 'order', 'position', 'depth', 'constant', 'disable'):
            text = f"{field_name}: {entry.get(field_name, '')}"
            detail_item = QTreeWidgetItem([text])
            detail_item.setData(0, ROLE_LEVEL, 4)
            entry_item.addChild(detail_item)

    def _left_text(self, entry: dict) -> str:
        """生成条目左栏摘要文本"""
        uid = entry.get('uid', '?')
        order = entry.get('order', '?')
        return f"{entry.get('comment', '')} uid={uid} order={order}"

    def save_expanded_state(self) -> set[int]:
        """保存当前展开的条目 UID（用于重建后恢复展开状态）。"""
        expanded: set[int] = set()
        it = QTreeWidgetItemIterator(self)
        while it.value():
            item = it.value()
            if item.data(0, ROLE_LEVEL) == 3 and item.isExpanded():
                uid = item.data(0, ROLE_UID)
                if uid is not None:
                    expanded.add(uid)
            ++it
        return expanded

    def restore_expanded_state(self, uids: set[int]):
        """恢复展开指定 UID 的条目节点。"""
        it = QTreeWidgetItemIterator(self)
        while it.value():
            item = it.value()
            if item.data(0, ROLE_LEVEL) == 3 and item.data(0, ROLE_UID) in uids:
                item.setExpanded(True)
            ++it
```

### 双栏 Delegate

```python
class DualColumnDelegate(QStyledItemDelegate):
    """
    将 QTreeWidgetItem 渲染为左右分栏。层级 4 条目不应用分栏（显示全文）。
    层级 3 条目：
    ┌─[☐]─🔵────────────────────────┬───────────────────────────────┐
    │ 左 45% (条目摘要)              │ 右 50% (映射信息)              │
    │ "学园都市 uid=1 order=1"       │ "→ 合并入 组0"                │
    └────────────────────────────────┴───────────────────────────────┘
    左侧 5% 留给 CheckBox（拆分模式显示），其后是条目类型图标。

    绘制规则：
    - level < 3: 不做分栏，默认绘制
    - level == 3: 执行分栏，检测 checkState + 图标 + 右栏着色
    - level == 4: 不做分栏，默认绘制（直接显示 "field: value"）

    图标约定（条目类型）：
    - 🔵 (蓝色圆圈) constant=true
    - 🟢 (绿灯) 动态条目
    - 📄 (文档) 混合/静态且非蓝灯
    """

    checkStateChanged = pyqtSignal(int, bool)  # uid, checked

    def paint(self, painter, option, index):
        level = index.data(ROLE_LEVEL)
        if level is None or level != 3:
            super().paint(painter, option, index)
            return

        painter.save()

        # 背景/选中态
        if option.state & QStyle.State_Selected:
            painter.fillRect(option.rect, option.palette.highlight())

        rect = option.rect
        w = rect.width()
        h = rect.height()
        left_w = w // 2
        right_w = w - left_w

        # 分割线
        painter.setPen(QPen(QColor("#ccc"), 1))
        painter.drawLine(rect.x() + left_w, rect.y(),
                         rect.x() + left_w, rect.y() + h)

        # 左栏：CheckBox 区域 + 图标 + 文本
        checkbox_w = 20
        icon_w = 20
        text_x = rect.x() + checkbox_w + icon_w + 4
        text_w = left_w - checkbox_w - icon_w - 8
        if index.data(ROLE_CHECKABLE):
            cb_rect = QRect(rect.x() + 2, rect.y(), checkbox_w, h)
            opt = QStyleOptionButton()
            opt.rect = cb_rect
            is_checked = (index.data(ROLE_CHECKED) == Qt.Checked)
            opt.state = (QStyle.State_On if is_checked else QStyle.State_Off)
            opt.state |= QStyle.State_Enabled
            self.parent().style().drawControl(QStyle.CE_CheckBox, opt, painter)
        # 图标
        icon_char = index.data(ROLE_ICON) or ""
        if icon_char:
            painter.setPen(QColor("#555"))
            painter.drawText(QRect(rect.x() + checkbox_w + 2, rect.y(), icon_w, h),
                             Qt.AlignCenter, icon_char)
        left_text = index.data(ROLE_LEFT_TEXT) or ""
        painter.setPen(option.palette.windowText().color())
        painter.drawText(QRect(text_x, rect.y(), text_w, h),
                         Qt.AlignLeft | Qt.AlignVCenter, left_text)

        # 右栏：根据映射类型着色背景
        right_text = index.data(ROLE_RIGHT_TEXT) or ""
        right_rect = QRect(rect.x() + left_w + 4, rect.y(), right_w - 4, h)
        if "合并" in right_text:
            bg = QColor("#E3F2FD")
        elif "拆分" in right_text:
            bg = QColor("#FFF9C4")
        elif "自动生成" in right_text:
            bg = QColor("#E8F5E9")
        else:
            bg = QColor("#F5F5F5")
        painter.fillRect(right_rect, bg)
        painter.setPen(QColor("#333"))
        painter.drawText(right_rect, Qt.AlignLeft | Qt.AlignVCenter, right_text)

        painter.restore()

    def editorEvent(self, event, model, option, index):
        """处理 CheckBox 点击事件（仅勾选框区域），发射 checkStateChanged 信号。"""
        if not index.data(ROLE_CHECKABLE):
            return False
        if event.type() == QEvent.MouseButtonRelease:
            cb_region = QRect(option.rect.x() + 2, option.rect.y(), 20, option.rect.height())
            if cb_region.contains(event.pos()):
                checked = (index.data(ROLE_CHECKED) == Qt.Checked)
                new_state = Qt.Unchecked if checked else Qt.Checked
                model.setData(index, new_state, ROLE_CHECKED)
                uid = index.data(ROLE_UID)
                if uid is not None:
                    self.checkStateChanged.emit(uid, new_state == Qt.Checked)
                return True
            # 勾选框区域外的点击不做任何处理，由树处理选择/展开
        return False
```

## 4. 同步滚动（全文 diff 模式）

使用单个 `QTextBrowser` 渲染 `difflib.HtmlDiff.make_file()` 生成的完整双栏 HTML，无需同步滚动。

```python
class FulltextDiffView(QWidget):
    """
    全文 diff 视图。
    difflib.HtmlDiff.make_file() 生成一个完整双栏 HTML 页面，
    渲染在单个 QTextBrowser 中。同步滚动由浏览器自身处理。
    """

    def __init__(self):
        self.browser = QTextBrowser()
        self.browser.setReadOnly(True)
        self.browser.setOpenExternalLinks(False)
        layout = QVBoxLayout()
        layout.addWidget(self.browser)
        self.setLayout(layout)

    def load_diff(self, left_text: str, right_text: str):
        differ = difflib.HtmlDiff(tabsize=2)
        html = differ.make_file(
            left_text.splitlines(keepends=True),
            right_text.splitlines(keepends=True),
            fromdesc="原始",
            todesc="优化后",
            context=True,
            numlines=3,
        )
        self.browser.setHtml(html)
```

## 5. 与管线的 subprocess 调用关系

所有 subprocess 调用均设置 `cwd=PROJECT_ROOT`，并传递 `PYTHONPATH=PROJECT_ROOT` 环境变量确保 `scripts/` 在模块搜索路径中。

约定临时目录：`{原文件所在目录}/.wb_optimize/`，对比模式自动检测此目录下的 `{文件名}_merged.json`。

GUI 分步调用脚本以实现进度追踪（优化模式下不直接调用 `run_pipeline`）：

## 6. 异步执行模型

### 6.1 Worker — 单命令执行

```python
from PyQt6.QtCore import QThread, pyqtSignal

class Worker(QThread):
    finished = pyqtSignal(str)        # stdout
    error = pyqtSignal(str, str)      # type ("timeout"|"non-zero"|"exception"), message

    def __init__(self, cmd: list[str], cwd: str | None = None, env: dict | None = None):
        super().__init__()
        self.cmd = cmd
        self.cwd = cwd
        self._env = env

    def run(self):
        try:
            result = subprocess.run(
                self.cmd, capture_output=True, text=True,
                cwd=self.cwd, env=self._env, timeout=120
            )
            if result.returncode != 0:
                self.error.emit("non-zero", result.stderr)
            else:
                self.finished.emit(result.stdout)
        except subprocess.TimeoutExpired:
            self.error.emit("timeout", "执行超时（120秒）")
        except Exception as e:
            self.error.emit("exception", str(e))
```

### 6.2 PipelineWorker — 5 步优化管线

```python
class PipelineWorker(QThread):
    """串行执行 5 步优化管线，支持进度追踪和取消。"""

    step_done = pyqtSignal(int, str)        # step_index (1-5), label
    step_error = pyqtSignal(int, str, str)  # step_index, type, message
    pipeline_finished = pyqtSignal(str)      # merged.json 路径
    pipeline_cancelled = pyqtSignal()

    def __init__(self, input_path: str, tmp_dir: str, wrapper_name: str = "补充内容"):
        super().__init__()
        self.input_path = input_path
        self.tmp_dir = tmp_dir
        self.wrapper_name = wrapper_name
        self._cancelled = False
        self._steps = [
            ("备份原文件", self._step_backup),
            ("分析条目", self._step_analyze),
            ("拆分条目", self._step_split),
            ("重排序", self._step_reorder),
            ("合并静态条目", self._step_merge),
        ]

    def cancel(self):
        self._cancelled = True

    def run(self):
        for i, (label, fn) in enumerate(self._steps, 1):
            if self._cancelled:
                self.pipeline_cancelled.emit()
                return
            try:
                fn()
                self.step_done.emit(i, label)
            except Exception as e:
                self.step_error.emit(i, "exception", str(e))
                return
        self.pipeline_finished.emit(os.path.join(self.tmp_dir, "_merged.json"))

    def _step_backup(self):
        import shutil, datetime
        backup_dir = os.path.join(os.path.dirname(self.input_path), "backup")
        os.makedirs(backup_dir, exist_ok=True)
        ts = datetime.datetime.now().strftime("%Y%m%d_%H%M%S")
        basename = os.path.basename(self.input_path)
        dest = os.path.join(backup_dir, f"{basename}.backup_{ts}.json")
        shutil.copy2(self.input_path, dest)

    def _run_cmd(self, cmd_args: list[str]):
        result = subprocess.run(
            cmd_args, capture_output=True, text=True,
            cwd=PROJECT_ROOT, timeout=120,
            env={**os.environ, "PYTHONPATH": str(PROJECT_ROOT)},
        )
        if result.returncode != 0:
            raise RuntimeError(result.stderr)

    def _step_analyze(self):
        analysis_out = os.path.join(self.tmp_dir, "_analysis.json")
        self._run_cmd([
            "python", "-m", "scripts.analyze_entries",
            "-i", self.input_path, "-o", analysis_out,
        ])

    def _step_split(self):
        analysis_in = os.path.join(self.tmp_dir, "_analysis.json")
        split_out = os.path.join(self.tmp_dir, "_split.json")
        self._run_cmd([
            "python", "-m", "scripts.split_entries",
            "-i", self.input_path, "-a", analysis_in,
            "-o", split_out, "-w", self.wrapper_name,
        ])

    def _step_reorder(self):
        split_in = os.path.join(self.tmp_dir, "_split.json")
        reorder_out = os.path.join(self.tmp_dir, "_reordered.json")
        self._run_cmd([
            "python", "-m", "scripts.reorder_entries",
            "-i", split_in, "-o", reorder_out,
            "-w", self.wrapper_name,
        ])

    def _step_merge(self):
        reorder_in = os.path.join(self.tmp_dir, "_reordered.json")
        merge_out = os.path.join(self.tmp_dir, "_merged.json")
        self._run_cmd([
            "python", "-m", "scripts.merge_entries",
            "-i", reorder_in, "-o", merge_out,
        ])
```
